```rust
pub struct ItiranScraperOutput {
  // 取得決算内容が含まれる
  pub items: Vec<EarningItems>,
}

pub struct EarningItems{
  // 未定。判別に使えるもの
```

## 新規判別後の出力(各サイト共通)(現在考えているサイトにおいては個別ページスクレイピング後)(notifyフローへ渡す最終正規化済み構造体)(monitorクレートで持つ)
```rust
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct MonitoredEarningsReport {
  // Jsonの仕様変更をしたとき、Rust側でスキーマを分ける用
  // 使わないが一応持たせる？
  pub schema_version: u32,

  // どのサイトで取得したかをラベル付けしておく
  pub source: String,
  
  // 取得時間(スクレイピング開始時間)
  pub fetched_at: DateTime<FixedOffset>,
  
  // 新規決算内容が含まれる。
  pub items: Vec<Earnings>,
}

// 決算内容
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Earnings {
  // 証券コード
  pub ticker: String,
  // 銘柄名
  pub company_name: String,
  // 公開時間
  pub published_at: DateTime<FixedOffset>,
  // 決算のタイトル
  pub title: String,
  // 詳細情報への
  pub url: String,
  // 内容の要約(速報情報)
  pub summary: String,
  // 決算評価
  pub evaluation: EarningsEvaluation,
}

pub enum EarningsEvaluation {
    Positive, // 正の評価
    Neutral,  // 通常
    Negative, // 負の評価
    Unrated,　// 評価なし
}

```
